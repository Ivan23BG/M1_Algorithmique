\begin{td-exo}[Forme standard et forme canonique]\,\\ % 12
	Dans cet exercice vous devez mettre les programmes suivants sous forme standard et donner
	également la forme matricielle.
	\begin{enumerate}
		\item
		\begin{equation*}
			\begin{cases}
				\max z = x_1 + x_2 \\
				x_1 + 5 x_2 \leq 5 \\
				2x_1 + x_2 \leq 4 \\
				x_1 \geq 0,\quad i=1,2.
			\end{cases}
		\end{equation*}

		\item
		\begin{equation*}
			\begin{cases}
				\max z = 80 x_1 + 60 x_2 \\
				0.2x_1 + 0.32 x_2 \leq 0.25\\
				x_1 + x_2 = 1 \\
				x_1 \geq 0,\quad i=1,2.
			\end{cases}
		\end{equation*}

		\item Réécrire le programme précédent dans le cas où la fonction objectif
		est la minimisation.

		\item
		\begin{equation*}
			\begin{cases}
				\max z = 5x_1 + 2x_2 \\
				6x_1 + x_2 \geq 6 \\
				4x_1 + 3x_2  \geq 12 \\
				x_1 + 2x_2 \geq 4 \\
				x_1 \geq 0,\quad i=1,2.
			\end{cases}
		\end{equation*}
	\end{enumerate}
\end{td-exo}

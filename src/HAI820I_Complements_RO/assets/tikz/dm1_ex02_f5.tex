}{
\fbox{
\begin{tikzpicture}[scale=1]

% ----------------------
% Parameters
% ----------------------

\def\controlRadius{1}          % control zones (cities 1-4)
\def\controlRadiusLarge{2}     % control zone (city 5)

\def\moatOuterRadius{1.5}      % outer moat boundary
\def\moatInnerRadius{1.485942} % inner moat boundary

\def\moatControlRadius{2.5}       % moat including control zones
\def\moatControlInnerRadius{2.485942}

% ----------------------
% Generic L1 ball
% ----------------------

\newcommand{\Lone}[2]{%
($ (#2)+(#1,0) $)
-- ++(-#1,#1)
-- ++(-#1,-#1)
-- ++(#1,-#1)
-- ++(#1,#1)
-- cycle
}

% ----------------------
% City coordinates
% ----------------------

\def\cities{
1/0/0,
2/3/0,
3/1/-1,
4/3/-2,
5/0/-3
}

% city labels
\foreach \i/\x/\y in \cities {
\node (\i) at (\x,\y) {\(\i\)};
}

% ----------------------
% Douves (moats)
% ----------------------

% Region {1,3}
\fill[fill=none, draw=green!70!pagetext, thick]
\foreach \v in {1,3} { \Lone{\moatOuterRadius}{\v} }
\Lone{\moatControlRadius}{5};

\fill[green!30, draw=none, thick]
\foreach \v in {1,3} { \Lone{\moatInnerRadius}{\v} }
\Lone{\moatControlInnerRadius}{5};

% Region {2,4}
\fill[fill=none, draw=green!70!pagetext, thick]
\foreach \v in {2,4} { \Lone{\moatOuterRadius}{\v} };

\fill[green!30, draw=none, thick]
\foreach \v in {2,4} { \Lone{\moatInnerRadius}{\v} };

% ----------------------
% Control zones
% ----------------------

\foreach \v in {1,2,3,4}{
\draw[red, thick, fill=red!20]
\Lone{\controlRadius}{\v};
}

\draw[red, thick, fill=red!20]
\Lone{\controlRadiusLarge}{5};

% ----------------------
% Legend
% ----------------------

\begin{scope}[shift={(5,0)}, font=\scriptsize]

\draw[red, thick, fill=red!20]
(0.25,0) -- ++(-0.25,0.25) -- ++(-0.25,-0.25)
-- ++(0.25,-0.25) -- ++(0.25,0.25) -- cycle;

\node[right=0.3cm] at (0,0) {Zones de contrôle};

\draw[green!70!pagetext, thick, fill=green!30]
(0.25,-0.7) -- ++(-0.25,0.25) -- ++(-0.25,-0.25)
-- ++(0.25,-0.25) -- ++(0.25,0.25) -- cycle;

\node[right=0.3cm] at (0,-0.7) {Douves};

\draw[blue, very thick]
(-0.25,-1.4) -- (0.25,-1.4);

\node[right=0.3cm] at (0,-1.4) {Solution optimale};

\end{scope}

% ----------------------
% Grid
% ----------------------

\foreach \y in {0,-1,-2,-3}{
\draw (0,\y) -- (3,\y);
}

\foreach \x in {0,1,2,3}{
\draw (\x,0) -- (\x,-3);
}

% ----------------------
% Redraw labels above shapes
% ----------------------

\foreach \i/\x/\y in \cities {
\node at (\x,\y) [anchor=south west] {\(\i\)};
}

% city dots
\foreach \i/\x/\y in \cities {
\draw[fill=pagetext] (\x,\y) circle (0.1);
}

% ----------------------
% Optimal solution
% ----------------------

\draw[blue, very thick]
(0,0) -- (3,0) -- (3,-3) -- (0,-3) -- (0, -2) -- (1,-2) -- (1,-1) -- (0,-1) -- cycle;

\end{tikzpicture}
}
}

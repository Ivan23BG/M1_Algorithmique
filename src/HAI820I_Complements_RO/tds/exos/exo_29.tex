\begin{td-exo}[Lagrangien]\,\\ % 29 
	Soit le problème \((P)\) suivant:
	\begin{equation*}
		(P) = 
		\begin{cases}
			\min f(x) = x_1^2 + x_2^2 \\
			g(x) = 2x_1 + x_2 + 4 \leq 0 \\
			x_i \in \bb R, \forall i \in \{1,2\}
		\end{cases}
	\end{equation*}
	\begin{enumerate}
		\item On relache la contrainte et on lui affecte le multiplicateur de Lagrange \(\lambda \geq 0\). 
		\item Enoncer \((D)\) le problème dual lagrangien de \((P)\).
		\item Calculer la fonction duale \(\Delta(\lambda)\) en fonction de \(\lambda\).
		\item Résoudre le problème dual \((D)\).
	\end{enumerate}
\end{td-exo}

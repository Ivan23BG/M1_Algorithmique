\begin{problem}{\textsc{Weighted Vertex Cover} (\(WVC\))}
    \Input{Un graphe \(G=(V,E,\omega)\) où \(\omega: V \to \bb R^+\) est une fonction de poids sur les sommets, et un entier \(k\).}
    \Question{Existe-t-il un ensemble de sommets \(V' \subseteq V\) qui couvre toutes les arêtes de \(G\) et dont le poids total est au plus \(k\)?}
\end{problem}

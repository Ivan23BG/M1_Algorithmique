\begin{algorithm}[H]
    \caption{\textsc{Exact Cover by 3-sets}}
    \begin{algorithmic}
        \Require{Un univers \(X\), une collection \(C\) de triplés de \(X\)}
        \Ensure{Trouver une collection \(C' \sub C\) telle que chaque élément de \(X\) apparait exactement une fois dans les triplés de \(C'\)}
    \end{algorithmic}
\end{algorithm}

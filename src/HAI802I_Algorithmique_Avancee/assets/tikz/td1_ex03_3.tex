\begin{problem}{\textsc{Steiner tree}}
    \Input{Un graphe \(G = (V, E)\), une fonction de poids \(w\) sur les arêtes, un ensemble de terminaux \(X \sub V\), un entier \(k\)}
    \Question{Trouver \(T \sub E\) tel que la somme des poids des arêtes de \(T\) soit au plus de \(k\), \(G[T]\) est connexe et \(X \sub V(G[T])\)}
\end{problem}

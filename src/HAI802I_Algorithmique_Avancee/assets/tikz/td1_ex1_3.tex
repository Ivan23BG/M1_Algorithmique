\begin{algorithm}[H]
    \caption{\(\lnot3\)-SAT}
    \begin{algorithmic}
        \Require{Une formule \(\varphi(x_1,\ldots,x_n)\) en 3-CNF}
        \Ensure{Existe-t-il une affectation des variables \((x_1,\ldots,x_n)\) qui ne satisfait pas \(\varphi\)?}
    \end{algorithmic}
\end{algorithm}

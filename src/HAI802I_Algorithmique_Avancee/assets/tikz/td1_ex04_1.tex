\vspace{-6mm}
\begin{algorithm}[H]
    \caption{\textsc{Coupe maximum} (CUT)}
    \begin{algorithmic}
        \Require{Soit \(G = (V, E)\) un graphe non orienté et \(k\in\bb N\).}
        \Ensure{Existe-t-il une partition des sommets en deux sous-ensembles \(V_1\) et \(V_2\) tels que le nombre d'arêtes entre \(V_1\) et \(V_2\) soit \(k\)?}
    \end{algorithmic}
\end{algorithm}

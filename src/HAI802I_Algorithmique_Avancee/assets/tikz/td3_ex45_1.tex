\begin{algorithm}[H]
    \caption{\textsc{Asymmetric TSP Problem (ATSP)}}
    \begin{algorithmic}
        \Require{Un graphe \(G = (V, E)\) orienté et muni d'une fonction de coût positive sur les arcs.}
        \Ensure{Trouver un circuit de cout minimum qui passe par chaque sommet exactement une fois.}
    \end{algorithmic}
\end{algorithm}
